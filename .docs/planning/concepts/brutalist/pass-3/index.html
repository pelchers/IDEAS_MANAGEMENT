$ cat /vault/schema/threat_intelligence.sql

          


          

            
CREATE TABLE threat_indicators (
    id              UUID            PRIMARY KEY DEFAULT gen_random_uuid(),
    indicator_type  VARCHAR(50)     NOT NULL CHECK (indicator_type IN ('ip','domain','hash','url','email')),
    value           TEXT            NOT NULL,
    severity        INTEGER         NOT NULL CHECK (severity BETWEEN 1 AND 10),
    source          VARCHAR(100)    NOT NULL,
    first_seen      TIMESTAMPTZ     NOT NULL DEFAULT now(),
    last_seen       TIMESTAMPTZ     NOT NULL DEFAULT now(),
    confidence      DECIMAL(3,2)    CHECK (confidence BETWEEN 0 AND 1),
    tags            TEXT[]          DEFAULT '{}',
    is_active       BOOLEAN         DEFAULT true
);

CREATE TABLE incident_reports (
    id              UUID            PRIMARY KEY DEFAULT gen_random_uuid(),
    title           VARCHAR(200)    NOT NULL,
    severity        VARCHAR(10)     NOT NULL CHECK (severity IN ('critical','high','medium','low','info')),
    status          VARCHAR(20)     NOT NULL DEFAULT 'open',
    reported_by     UUID            REFERENCES analysts(id),
    assigned_to     UUID            REFERENCES analysts(id),
    created_at      TIMESTAMPTZ     NOT NULL DEFAULT now(),
    resolved_at     TIMESTAMPTZ,
    mitre_techniquesTEXT[]          DEFAULT '{}',
    evidence_refs   JSONB           DEFAULT '[]'
);

CREATE TABLE analysts (
    id              UUID            PRIMARY KEY DEFAULT gen_random_uuid(),
    handle          VARCHAR(50)     UNIQUE NOT NULL,
    clearance       VARCHAR(20)     NOT NULL CHECK (clearance IN ('L1','L2','L3','admin')),
    specialization  VARCHAR(100),
    active          BOOLEAN         DEFAULT true,
    last_login      TIMESTAMPTZ
);

CREATE INDEX idx_indicators_type ON threat_indicators(indicator_type);
CREATE INDEX idx_indicators_severity ON threat_indicators(severity DESC);
CREATE INDEX idx_incidents_status ON incident_reports(status) WHERE status != 'resolved';


          


          

            
-- 3 tables, 4 indexes, 2 foreign keys
